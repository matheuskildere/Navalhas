//SQL TABELA: BARBEIRO 

-- Table: public.barbeiro

-- DROP TABLE public.barbeiro;

CREATE TABLE public.barbeiro
(
    id integer NOT NULL,
    nome character(50) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT barbeiro_pkey PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE public.barbeiro
    OWNER to postgres;

//SQL TABELA: CLIENTE

-- Table: public.cliente

-- DROP TABLE public.cliente;

CREATE TABLE public.cliente
(
    id integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 0 MINVALUE 0 MAXVALUE 2000 CACHE 1 ),
    nom character(100) COLLATE pg_catalog."default",
    cont1 character(20) COLLATE pg_catalog."default",
    cont2 character(20) COLLATE pg_catalog."default",
    foto character(250) COLLATE pg_catalog."default",
    servgratis integer,
    servtotal integer,
    serv integer,
    bio smallint,
    CONSTRAINT cliente_pkey PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE public.cliente
    OWNER to postgres;

//SQL TABELA: SERVICO

-- Table: public.servico

-- DROP TABLE public.servico;

CREATE TABLE public.servico
(
    id integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 0 MINVALUE 0 MAXVALUE 10000 CACHE 1 ),
    valtotal double precision NOT NULL,
    idcli integer NOT NULL,
    idbar integer NOT NULL,
    paga boolean NOT NULL,
    data character(15) COLLATE pg_catalog."default",
    CONSTRAINT servico_pkey PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE public.servico
    OWNER to postgres;
